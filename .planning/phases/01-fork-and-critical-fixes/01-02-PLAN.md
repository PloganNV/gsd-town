       ```go
       // runDonePreFlight checks commit/push/PR state before marking a bead complete.
       // In auto-fix mode (default), it pushes and creates a PR if missing.
       // In strict mode (--strict flag), it returns an error instead of auto-fixing.
       // In force mode (--force flag), all checks are skipped.
       //
       // CHAR(36) UUID note: beads events.id is a string, not int64 (GH#3622).
       func runDonePreFlight(g *git.Git, branch, defaultBranch, issueID string, cwd string, strict, force bool) error {
           if force {
               return nil
           }

           // Check push status
           pushed, _, err := g.BranchPushedToRemote(branch, "origin")
           if err != nil {
               style.PrintWarning("pre-flight: could not check push status: %v (skipping push check)", err)
               pushed = true // assume pushed on error to avoid blocking
           }

           if !pushed {
               if strict {
                   return fmt.Errorf("pre-flight: branch %q not pushed to origin (use --force to bypass or push manually)", branch)
               }
               // Auto-push
               refspec := branch + ":" + branch
               if pushErr := g.Push("origin", refspec, false); pushErr != nil {
                   return fmt.Errorf("pre-flight: auto-push failed: %w\nPush manually or use --force to bypass", pushErr)
               }
               fmt.Printf("%s Pre-flight: branch pushed to origin/%s\n", style.Bold.Render("✓"), branch)
           }

           // Check PR existence (requires gh CLI)
           ghPath, lookErr := exec.LookPath("gh")
           if lookErr != nil {
               style.PrintWarning("pre-flight: gh CLI not found — skipping PR check")
               return nil
           }
           _ = ghPath

           prNum, prErr := checkPreFlightPR(branch)
           if prErr != nil {
               style.PrintWarning("pre-flight: could not check PR status: %v (skipping PR check)", prErr)
               return nil
           }
           if prNum > 0 {
               return nil // PR already exists
           }

           // No PR found
           if strict {
               return fmt.Errorf("pre-flight: no open PR found for branch %q (use --force to bypass or create PR manually)", branch)
           }

           // Auto-create PR
           createdNum, createErr := createPreFlightPR(branch, defaultBranch, issueID)
           if createErr != nil {
               return fmt.Errorf("pre-flight: auto-PR creation failed: %w\nCreate PR manually or use --force to bypass", createErr)
           }
           fmt.Printf("%s Pre-flight: PR #%d created for branch %s\n", style.Bold.Render("✓"), createdNum, branch)
           return nil
       }

       // checkPreFlightPR returns the PR number (0 if none) for an open PR on branch.
       func checkPreFlightPR(branch string) (int, error) {
           cmd := exec.Command("gh", "pr", "list",
               "--head", branch, "--json", "number", "--state", "open")
           out, err := cmd.Output()
           if err != nil {
               return 0, fmt.Errorf("gh pr list: %w", err)
           }
           var prs []struct{ Number int `json:"number"` }
           if err := json.Unmarshal(out, &prs); err != nil {
               return 0, fmt.Errorf("parse pr list: %w", err)
           }
           if len(prs) == 0 {
               return 0, nil
           }
           return prs[0].Number, nil
       }

       // createPreFlightPR creates a PR for branch → defaultBranch.
       func createPreFlightPR(branch, defaultBranch, issueID string) (int, error) {
           title := fmt.Sprintf("fix: %s", branch)
           if issueID != "" {
               title = fmt.Sprintf("fix: %s", issueID)
           }
           body := ""
           if issueID != "" {
               body = fmt.Sprintf("Closes %s", issueID)
           }
           args := []string{"pr", "create",
               "--head", branch, "--base", defaultBranch,
               "--title", title, "--body", body, "--json", "number"}
           cmd := exec.Command("gh", args...)
           out, err := cmd.Output()
           if err != nil {
               return 0, fmt.Errorf("gh pr create: %w", err)
           }
           var result struct{ Number int `json:"number"` }
           if err := json.Unmarshal(out, &result); err != nil {
               return 0, fmt.Errorf("parse pr create: %w", err)
           }
           return result.Number, nil
       }
       ```

_ensure_dep() {

Function signature:

bootstrap_town() {
  # Args:
  #   $1 — project_dir: absolute path to current GSD project root
  #   $2 — rig_name: slugified project name (e.g., "my-project")
  # Returns: 0 on success, 1 on failure
  # Outputs: progress messages on stdout, errors on stderr

Implementation steps (in order):

Step 1 — Create managed town if not present:
```bash
if [ ! -d "${GSD_TOWN_ROOT}/gastown" ]; then
  echo "  [bootstrap] creating managed town at ${GSD_TOWN_ROOT}..."
  _local_gt install "$GSD_TOWN_ROOT" --git 2>&1 || return 1
fi
```
Where _local_gt is a local inline function: `command -v gt >/dev/null 2>&1 && gt "$@" || "${HOME}/.local/bin/gt" "$@"`

Step 2 — Start the daemon (gt up):
```bash
echo "  [bootstrap] starting daemon..."
(cd "$GSD_TOWN_ROOT" && PATH="$PATH:${HOME}/go/bin:/opt/homebrew/bin" _local_gt up 2>&1) || return 1
```

Step 3 — Register project as rig (idempotent):
```bash
local existing_rigs
existing_rigs=$(cd "$GSD_TOWN_ROOT" && _local_gt rig list --json 2>/dev/null || echo "[]")
if ! echo "$existing_rigs" | python3 -c "import sys,json; rigs=json.load(sys.stdin); exit(0 if any(r.get('name')=='${rig_name}' for r in rigs) else 1)" 2>/dev/null; then
  echo "  [bootstrap] adding rig: $rig_name -> $project_dir"
  (cd "$GSD_TOWN_ROOT" && _local_gt rig add "$rig_name" "$project_dir" 2>&1) || return 1
else
  echo "  [bootstrap] rig $rig_name already registered"
fi
```

Step 4 — Add crew member (idempotent):
```bash
local username
username=$(git config user.name 2>/dev/null | tr ' ' '-' | tr '[:upper:]' '[:lower:]' || echo "${USER:-polecat}")
local existing_crew
existing_crew=$(cd "$GSD_TOWN_ROOT" && _local_gt crew list --json 2>/dev/null || echo "[]")
if ! echo "$existing_crew" | python3 -c "import sys,json; crew=json.load(sys.stdin); exit(0 if any(c.get('name')=='${username}' for c in crew) else 1)" 2>/dev/null; then
  echo "  [bootstrap] adding crew member: $username"
  (cd "$GSD_TOWN_ROOT" && _local_gt crew add "$username" 2>&1) || return 1
else
  echo "  [bootstrap] crew member $username already present"
fi
```

Step 5 — Update GSD_TOWN_RIG_DIR to reflect the rig we just created:
```bash
GSD_TOWN_RIG_DIR="${GSD_TOWN_ROOT}/${rig_name}/mayor/rig"
echo "  [bootstrap] rig dir: $GSD_TOWN_RIG_DIR"
```

Step 6 — Write ~/.gsd-town-config if it doesn't exist (lightweight registry file):
```bash
local config_file="${HOME}/.gsd-town-config"
if [ ! -f "$config_file" ]; then

  echo "  [bootstrap] wrote config: $config_file"
fi
```
